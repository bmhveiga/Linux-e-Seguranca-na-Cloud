# Comandos usados - Tópico 04

# Verificar serviço web
systemctl status nginx --no-pager

# Verificar portas em escuta
sudo ss -tulpn | grep -E ':(80|443|2222)'

# Verificar firewall
sudo ufw status

# Permitir portas necessárias
sudo ufw allow 80/tcp
sudo ufw allow 443/tcp
sudo ufw allow 2222/tcp

# Validar configuração do Nginx
sudo nginx -t

# Recarregar Nginx
sudo systemctl reload nginx

# Validar HTTP
curl -I http://localhost

# Validar HTTPS com certificado self-signed
curl -k -I https://localhost

# Verificar configuração efetiva do SSH
sudo sshd -T | grep -Ei 'port|permitrootlogin|passwordauthentication|pubkeyauthentication'

# Verificar permissões da raiz web
ls -ld /var/www/html
ls -l /var/www/html
